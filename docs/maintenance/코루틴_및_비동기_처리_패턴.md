# 코루틴 및 비동기 처리 패턴

## 개요

Unity 코루틴을 활용한 비동기 처리 패턴과 타임아웃 메커니즘을 분석합니다.

## Unity 코루틴 기본

### 코루틴 시작

```csharp
// MelonLoader에서 코루틴 시작
MelonCoroutines.Start(SomeCoroutine());

// 코루틴 메서드
private static IEnumerator SomeCoroutine()
{
    // 작업 수행

    // 다음 작업
}
```

## BgmBgaInjector - 주기적 체크

### 주입 코루틴

```csharp
public static class BgmBgaInjector
{
    private static Coroutine _injectionCoroutine = null;
    private static bool _isPlayScene = false;

    public static void StartInjection(bool isPlayScene)
    {
        _isPlayScene = isPlayScene;
        
        if (_injectionCoroutine != null)
        {
            MelonCoroutines.Stop(_injectionCoroutine);
        }
        
        _injectionCoroutine = MelonCoroutines.Start(InjectBgmBgaCoroutine());
    }

    private static IEnumerator InjectBgmBgaCoroutine()
    {
        while (true)
        {
            yield return new WaitForSeconds(2.0f); // 2초마다 체크
            
            try
            {
                // BGA 주입 시도
                if (!string.IsNullOrEmpty(_currentBgaFile))
                {
                    MelonCoroutines.Start(BgaInjector.TryInjectBgaCoroutine(_currentBgaFile));
                }
                
                // BGM 주입 시도 (플레이 씬에서만)
                if (_isPlayScene && !string.IsNullOrEmpty(_currentBgmFile))
                {
                    MelonCoroutines.Start(BgmInjector.TryInjectBgmCoroutine(_currentBgmFile));
                }
            }
            catch (Exception ex)
            {
                ErrorLogger.LogException(ex, "[BgmBgaInjector]", "주입 코루틴 오류");
            }
        }
    }

    public static void StopInjection()
    {
        if (_injectionCoroutine != null)
        {
            MelonCoroutines.Stop(_injectionCoroutine);
            _injectionCoroutine = null;
        }
    }
}
```

## BgaInjector - 타임아웃 처리

### BGA 주입 코루틴

```csharp
public static IEnumerator TryInjectBgaCoroutine(string bgaFilePath)
{
    if (string.IsNullOrEmpty(bgaFilePath) || !File.Exists(bgaFilePath))
    {
        yield break;
    }

    // 파일 크기 확인 및 타임아웃 동적 조정
    var fileInfo = new FileInfo(bgaFilePath);
    var fileSizeMB = fileInfo.Length / (1024.0 * 1024.0);
    
    // 100MB당 1초 추가, 최대 60초
    var timeout = Math.Min(10 + (int)(fileSizeMB / 100), 60);
    
    MelonLogger.Msg($"[BgaInjector] BGA 주입 시작 (크기: {fileSizeMB:F2}MB, 타임아웃: {timeout}초)");

    // VideoPlayer 찾기
    var videoPlayers = UnityEngine.Object.FindObjectsOfType<UnityEngine.Video.VideoPlayer>();
    if (videoPlayers.Length == 0)
    {
        yield break;
    }

    foreach (var videoPlayer in videoPlayers)
    {
        if (!videoPlayer.gameObject.activeInHierarchy)
            continue;

        try
        {
            // URL 설정
            var uri = new Uri(bgaFilePath).AbsoluteUri;
            videoPlayer.url = uri;
            
            // Prepare 시작
            videoPlayer.Prepare();
            
            // Prepare 완료 대기 (타임아웃 적용)
            float elapsedTime = 0f;
            while (!videoPlayer.isPrepared && elapsedTime < timeout)
            {
                yield return new WaitForSeconds(0.1f);
                elapsedTime += 0.1f;
            }

            if (!videoPlayer.isPrepared)
            {
                MelonLogger.Warning($"[BgaInjector] Prepare 타임아웃 ({timeout}초)");
                continue;
            }

            // 재생 시작
            videoPlayer.Play();
            
            // BGA/BGM 동기화 시작
            BgaBgmSyncManager.StartSync(videoPlayer);
            
            MelonLogger.Msg("[BgaInjector] BGA 주입 성공!");
        }
        catch (Exception ex)
        {
            ErrorLogger.LogException(ex, "[BgaInjector]", "BGA 주입 중 오류");
        }
    }
}
```

## BgmLoader - 오디오 로딩

### 비동기 오디오 로딩

```csharp
public static IEnumerator LoadAndInjectAudioClip(string bgmFilePath, object bgmManager)
{
    if (string.IsNullOrEmpty(bgmFilePath) || !File.Exists(bgmFilePath))
    {
        yield break;
    }

    // 파일 크기 확인 및 타임아웃 동적 조정
    var fileInfo = new FileInfo(bgmFilePath);
    var fileSizeMB = fileInfo.Length / (1024.0 * 1024.0);
    
    // 10MB당 1초 추가, 최대 60초
    var timeout = Math.Min(5 + (int)(fileSizeMB / 10), 60);
    
    MelonLogger.Msg($"[BgmLoader] BGM 로딩 시작 (크기: {fileSizeMB:F2}MB, 타임아웃: {timeout}초)");

    // AudioType 결정
    var audioType = GetAudioType(bgmFilePath);
    var uri = new Uri(bgmFilePath).AbsoluteUri;

    // UnityWebRequest로 오디오 로드
    using (var www = UnityWebRequestMultimedia.GetAudioClip(uri, audioType))
    {
        www.timeout = timeout;
        www.SendWebRequest();

        // 로딩 완료 대기
        while (!www.isDone)
        {
            yield return new WaitForSeconds(0.1f);
        }

        if (www.result != UnityWebRequest.Result.Success)
        {
            MelonLogger.Error($"[BgmLoader] BGM 로드 실패: {www.error}");
            yield break;
        }

        var audioClip = DownloadHandlerAudioClip.GetContent(www);
        if (audioClip == null)
        {
            MelonLogger.Error("[BgmLoader] AudioClip이 null입니다.");
            yield break;
        }

        // setClip 호출
        var setClipMethod = ReflectionHelper.FindMethod(
            "IntiCreates.cBGMBeatManager", "setClip");
        
        if (setClipMethod != null)
        {
            setClipMethod.Invoke(bgmManager, new object[] { audioClip });
            
            // 주입 검증
            if (VerifyInjection(bgmManager, audioClip))
            {
                MelonLogger.Msg("[BgmLoader] BGM 주입 성공!");
                
                // 게임 종료 시간 설정
                BgmFinishTimeManager.SetFinishTime(audioClip.length);
                
                // 재생 시작
                RequestPlayAudio(bgmManager);
            }
        }
    }
}
```

## BgaBgmSyncManager - 동기화

### 동기화 코루틴

```csharp
public static class BgaBgmSyncManager
{
    private static Coroutine _syncCoroutine = null;
    private static UnityEngine.Video.VideoPlayer _currentVideoPlayer = null;

    public static void StartSync(UnityEngine.Video.VideoPlayer videoPlayer)
    {
        _currentVideoPlayer = videoPlayer;
        
        // 즉시 동기화
        SyncBgaToBgm(videoPlayer);
        
        // 지속적인 동기화 시작
        if (_syncCoroutine != null)
        {
            MelonCoroutines.Stop(_syncCoroutine);
        }
        
        _syncCoroutine = MelonCoroutines.Start(SyncCoroutine());
    }

    private static IEnumerator SyncCoroutine()
    {
        while (_currentVideoPlayer != null)
        {
            yield return new WaitForSeconds(0.1f); // 0.1초마다 체크
            
            try
            {
                if (_currentVideoPlayer.isPlaying)
                {
                    var bgmTime = GetBgmTime();
                    var bgaTime = _currentVideoPlayer.time;
                    
                    // 0.1초 이상 차이나면 재동기화
                    if (Math.Abs(bgmTime - bgaTime) > 0.1)
                    {
                        MelonLogger.Msg($"[BgaBgmSyncManager] 재동기화 (차이: {Math.Abs(bgmTime - bgaTime):F3}초)");
                        SyncBgaToBgm(_currentVideoPlayer);
                    }
                }
            }
            catch (Exception ex)
            {
                ErrorLogger.LogWarning(ex, "[BgaBgmSyncManager]", "동기화 체크 중 오류");
            }
        }
    }

    private static void SyncBgaToBgm(UnityEngine.Video.VideoPlayer videoPlayer)
    {
        try
        {
            var bgmTime = GetBgmTime();
            videoPlayer.time = bgmTime;
            
            MelonLogger.Msg($"[BgaBgmSyncManager] BGA 동기화: {bgmTime:F3}초");
        }
        catch (Exception ex)
        {
            ErrorLogger.LogWarning(ex, "[BgaBgmSyncManager]", "동기화 실패");
        }
    }

    public static void StopSync()
    {
        if (_syncCoroutine != null)
        {
            MelonCoroutines.Stop(_syncCoroutine);
            _syncCoroutine = null;
        }
        
        _currentVideoPlayer = null;
    }
}
```

## BgmMonitorCoroutine - BGM 모니터링

### 게임 종료 모니터링

```csharp
public static class BgmMonitorCoroutine
{
    private static Coroutine _monitorCoroutine = null;

    public static void StartBgmMonitorCoroutine(object bgmManager, float bgmLength)
    {
        if (_monitorCoroutine != null)
        {
            MelonCoroutines.Stop(_monitorCoroutine);
        }
        
        _monitorCoroutine = MelonCoroutines.Start(MonitorBgmFinishCoroutine(bgmManager, bgmLength));
    }

    private static IEnumerator MonitorBgmFinishCoroutine(object bgmManager, float bgmLength)
    {
        MelonLogger.Msg($"[BgmMonitorCoroutine] BGM 모니터링 시작 (길이: {bgmLength:F2}초)");
        
        while (true)
        {
            yield return new WaitForSeconds(0.1f); // 0.1초마다 체크
            
            try
            {
                // BGM 재생 시간 확인
                var currentTime = GetCurrentBgmTime(bgmManager);
                
                // BGM 길이 도달 시 게임 종료
                if (currentTime >= bgmLength - 0.1f)
                {
                    MelonLogger.Msg($"[BgmMonitorCoroutine] BGM 종료 감지 ({currentTime:F2}초)");
                    
                    // 게임 종료 요청
                    RequestGameEnd();
                    
                    break;
                }
            }
            catch (Exception ex)
            {
                ErrorLogger.LogWarning(ex, "[BgmMonitorCoroutine]", "모니터링 중 오류");
            }
        }
        
        _monitorCoroutine = null;
    }

    public static void StopMonitoring()
    {
        if (_monitorCoroutine != null)
        {
            MelonCoroutines.Stop(_monitorCoroutine);
            _monitorCoroutine = null;
        }
    }
}
```

## 재시도 패턴

### 최대 재시도 횟수

```csharp
public static IEnumerator TryInjectBgmCoroutine(string bgmFilePath)
{
    const int maxAttempts = 10;
    int attempts = 0;
    
    while (attempts < maxAttempts)
    {
        attempts++;
        
        // BGM 매니저 찾기
        var bgmManager = FindBgmManager();
        if (bgmManager == null)
        {
            yield return new WaitForSeconds(1.0f);
            continue;
        }
        
        // 주입 시도
        yield return LoadAndInjectAudioClip(bgmFilePath, bgmManager);
        
        // 성공 확인
        if (VerifyInjection(bgmManager))
        {
            MelonLogger.Msg($"[BgmInjector] BGM 주입 성공 ({attempts}번째 시도)");
            yield break;
        }
        
        yield return new WaitForSeconds(1.0f);
    }
    
    MelonLogger.Warning($"[BgmInjector] BGM 주입 실패 ({maxAttempts}번 시도)");
}
```
